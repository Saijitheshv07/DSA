# [242. Valid Anagram](https://leetcode.com/problems/valid-anagram/description/)

- Basic freq array  / sorting usecase

### C++ Solution

```cpp
//Using freq arary
class Solution {
public:
    bool isAnagram(string s, string t) {
        int count[26] = {0};
        for (char x : s) {
            count[x - 'a']++;
        }
        for (char x : t) {
            count[x - 'a']--;
        }
        for (int val : count) {
            if (val != 0) {
                return false;
            }
        }
        return true;
    }
};

//Using sorting
class Solution {
public:
    bool isAnagram(string s, string t) {
        sort(s.begin(), s.end());
        sort(t.begin(), t.end());
        return s == t;
    }
};
```


### Java Solution

//Using freq array
class Solution {
    public boolean isAnagram(String s, String t) {
        if(s.length() != t.length()) return false;
        int[] arr = new int[26];

        for(int i=0;i<s.length();i++){
            arr[s.charAt(i) - 'a']++;
            arr[t.charAt(i) - 'a']--;
        }

        for(int i=0;i<26;i++){
            if(arr[i] != 0) return false;
        }
        return true;
    }
}

import java.util.Arrays;
class Solution {
    public boolean isAnagram(String s, String t) {
        char[] sChars = s.toCharArray();
        char[] tChars = t.toCharArray();
        
        Arrays.sort(sChars);
        Arrays.sort(tChars);
        
        return Arrays.equals(sChars, tChars);
    }
}
